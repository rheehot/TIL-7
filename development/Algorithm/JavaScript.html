  • 쇠막대기
    • '(' 을 push, ')'은 pop
    • ')'이 pop 될 때, 레이저인지 막대기인지 파악한다.
    • 레이저인 경우 '(' : 스택의 길이 =(현재까지 쌓인 막대길이)를 add 하고, 막대기 끝인 경우 ')' 1 을 add 한다.

      function solution(arrangement) {
    let answer = 0;
    const stack = [];
    for(let i = 0; i<arrangement.length; i++) {
        if(arrangement[i] === '(') {
            stack.push('(');
        } else {
            stack.pop();
            if (arrangement[i-1] === '(') {
                answer += stack.length;
            }
            else {
                answer += 1
            }
        }
    }
    return answer;
}
